 30   var pointCoordinates = [],
 31       point = [];
 32   //go through coordinates
 33   for(var i = 0; i < route.coordinates.length; i++) {
 34     point = route.coordinates[i];
 35     //push them to coordinate array
 36     pointCoordinates.push([point.lng, point.lat]);
 37   }
 38   //convert to GeoJson
 39   var string = toGeoJson(pointCoordinates);
 40   //save to localstorage
 41   localStorage.setItem("storageRoutes", JSON.stringify(string));
 42   //output into textarea
 43   document.getElementById("GeoJson").value = JSON.stringify(string);
 44 //also add to map
 45 }).addTo(map);

 47 /**
 48   * converts an array of coordinates into a LineString in GeoJSON format
 49   * done by using this source:
 50   * @see https://github.com/perliedman/leaflet-routing-machine/blob/344ff09c8bb94d4e42fa583286d95396d8227c65/src/L.Routing.js
 51   */
 52 function toGeoJson(coordinates) {
 53   return {
 54     type: 'FeatureCollection',
 55     features: [
 56       {
 57         type: 'Feature',
 58         properties: {},
 59         geometry: {
 60           type: 'LineString',
 61           coordinates: coordinates
 62         }
 63       },
 64     ]
 65   };
 66 }

 68 function fillDate() {
 69   var now = new Date();
 70   let year = now.getFullYear();
 71   let month = '' + (now.getMonth() + 1);
 72   if(month.length < 2) { month = '0' + month; }
 73   let day = '' + now.getDate();
 74   if(day.length < 2) { day = '0' + day; }
 75   now = day+'-'+month+'-'+year;
 76   document.getElementById("routeDate").value = now;
 77 }

 79 /**
 80   * function that handles the createButton event
 81   */
 82 function createRouteButton() {
 83   event.preventDefault();
 84 
 85   var newRoute = document.getElementById("GeoJson").value;
 86   var addAttr = JSON.parse(newRoute);
 87   addAttr.name = document.getElementById("routeName").value;
 88   addAttr.routeType = document.querySelector('input[name="routeType"]:checked').value;
 89   addAttr.username = document.getElementById("routeUsername").value;
 90   addAttr.date = document.getElementById("routeDate").value;
 91   addAttr.desc = document.getElementById("routeDesc").value;
 92   var newRouteString = JSON.stringify(addAttr);
 93 
 94   //post
 95   $.ajax({
 96     type: 'POST',
 97     data: newRouteString,
 98     url: '/users/addroute',
 99     contentType:"application/json"
100   }).done(function(response) {
101     console.dir(response);
102     //successful
103     if(response.error === 0) {
104       alert('route added: ' + response.msg);
105     }
106     else {
107       alert('Error: ' + response.msg);
108     }
109   });
110 }
