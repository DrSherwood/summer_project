 33 /**
 34   *@desc  function gets the routes from the database
 35   * and creates a table according to each entry
 36   *@param changer has a value from a select form
 37   *and changes the output of the table
 38   *@author Benjamin Rieke
 39   */
 40   function readRoutesSelect(changer) {
 41 
 42   //get JSON from DB
 43   $.getJSON('/users/routes', function(data) {
 44     routesJSON = data;
 45     var tableContent ='';
 46     $.each(data, function(index) {
 47     if (routesJSON[index].features[0].geometry.type  == "LineString" && routesJSON[index].routeType == changer  ) {
 48       tableContent += '<tr>';
 49       var checkbox = "<input type='checkbox' id='route"+index+"' name='routes' onchange='displayRoute("+index+")'></checkbox>";
 50       tableContent += '<td>' + checkbox + '</td>';
 51       tableContent += '<td>' + this.name + '</td>';
 52       tableContent += '<td>' + this.routeType + '</td>';
 53       tableContent += '<td>' + this.username + '</td>';
 54       tableContent += '<td>' + this.date + '</td>';
 55       tableContent += '</tr>';
 56 }
 57 if (changer == "Type") {
 58   readRoutes();
 59 }
 60 
 61     });
 62 
 63     $('#resultTable').html(tableContent);
 64   });
 65 }

103 function displayRoute(index) {
104   console.log("displayRoute "+index);
105   var id = "route";
106   id += index;
107   if(document.getElementById(id).checked) {
108     //put into routeInQuestion textarea
109     document.getElementById("routeInQuestion").value = JSON.stringify(routesJSON[index]);
110     document.getElementById("p1").innerHTML = "You chose:   " + JSON.stringify(routesJSON[index].name);
111 
112 
113     var LatLon = turnLatLon(routesJSON[index].features[0].geometry.coordinates);
114     var line = L.polyline(LatLon, {color: "red", weight: 4});
115     line.addTo(map);
116     map.fitBounds(line.getBounds());
117     //insert into 'memory' array
118     routesToDelete[index] = line;
119   }
120   else {
121     console.log("remove");
122     map.removeLayer(routesToDelete[index]);
123   }
124 }

126 /**
127   * quick function to turn around the coordinates
128   * @param array
129   */
130 function turnLatLon(array) {
131   var latlon = [];
132   //go through points and change to latlon order
133   for (var j = 0; j < array.length; j++) {
134     var point = [];
135     var lat = array[j][1];
136     point.push(lat);
137     var lon = array[j][0];
138     point.push(lon);
139     latlon.push(point);
140   }
141   return latlon;
142 }
