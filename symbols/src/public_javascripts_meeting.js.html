 70 /**
 71   * @desc Checks if two choosen routes are intersecting eachother
 72   * Uses turf.js
 73   * @see https://turfjs.org/
 74   * @param firstLine The chosen route that gets checked for intersections
 75   * @param secondLine Another route that gets checked for intersections with the first one
 76   * @author Benjamin Rieke 408743
 77   */
 78 var counter = 0;
 79 function lineStringsIntersect(firstLine, secondLine) {
 80   console.log("lSI called with 1: " + JSON.parse(firstLine).name + " 2: " + JSON.parse(secondLine).name);
 81   //lines geometry
 82   var l1 = JSON.parse(firstLine);
 83   var l2 = JSON.parse(secondLine);
 84   //console.log(l1);
 85   //GeoJSON of intersections
 86   var intersects = turf.lineIntersect(l1.features[0].geometry, l2.features[0].geometry);
 87   // for each intersection do
 88   for(let i = 0; intersects.features[i]; i++) {
 89 
 90     intersects.features[i].properties.firstUsername = l1.username;
 91     intersects.features[i].properties.secondUsername = l2.username;
 92     intersects.features[i].properties.firstID = l1._id;
 93     intersects.features[i].properties.secondID = l2._id;
 94     intersects.features[i].properties.firstType = l1.routeType;
 95     intersects.features[i].properties.secondType = l2.routeType;
 96     intersects.features[i].properties.intersectNumber = counter;
 97     intersects.features[i].properties.intersectRootRoute = l2.features[0].geometry.coordinates;
 98 
 99     //increase the counter to keep track
100     counter ++;
101 
102     //console.log(JSON.stringify(intersects.features[i]));
103     var meetingPoint = (intersects.features[i].geometry.coordinates);
104     // switch lat and long for display on leaflet
105     intersects.features[i].geometry.coordinates = [meetingPoint[1], meetingPoint[0]];
106     //push to meetingsArray
107     meetingsArray.push(intersects.features[i]);
108     let thisPoint = intersects.features[i];
109 // print the interscetions into a table
110     intersectOutput();
111 // get the weather for each individual intersection
112     xhrGetWeather(thisPoint.geometry.coordinates[0], thisPoint.geometry.coordinates[1], handleWeather, thisPoint, meetingsArray);
113 
114   }
115 }

119 /**
120   * this function handles the xhr request to openweathermap
121   * @author Joanthan Bahlmann
122   * @param lat Latitude
123   * @param lon Longitude
124   * @param cFunc callback function for successful request
125   * @param i index passed down into cFunc
126   */
127 function xhrGetWeather(lat, lon, cFunc, meetingPoint, meetingsArray) {
128   console.log("xhrGetWeather called");
129   var url = "https://api.openweathermap.org/data/2.5/weather?lat="+lat+"&lon="+lon+"&appid="+apiKey;
130   var xhttp = new XMLHttpRequest();
131   //when ready
132   xhttp.onreadystatechange = function() {
133     if(this.readyState == 4 && this.status == 200) {
134       cFunc(this, meetingPoint);
135     }
136   };
137   xhttp.open("GET", url, true);
138   xhttp.send();
139   console.log("xhr send");
140 }

142 /**
143   * this function handles the xhttp response from OWMTOKEN
144   * it saves the weather into a table, creates a marker at the found weather position
145   * also the additional information xhr requests are called from here, depending on the weather code
146   * @author Jonathan Bahlmann
147   * @param xhttp xhttp responseText
148   * @param i index to save the responseText at the right place in table
149   */
150 function handleWeather(xhttp, meetingPoint, meetingsArray) {
151   //parse resonse
152   weatherXML = JSON.parse(xhttp.responseText);
153   let temperature = weatherXML.main.temp;
154   let toCelsius = 273.1;
155   temperature = temperature - toCelsius;
156   temperature = Math.round(temperature*10)/10;
157   let wString = '';
158   wString += weatherXML.weather[0].description + ', ' + temperature + '°C';
159 
160   var n1 = meetingPoint.properties.firstUsername;
161   var n2 = meetingPoint.properties.secondUsername;
162   var t1 = meetingPoint.properties.firstType;
163   var t2 = meetingPoint.properties.secondType;
164   var meetNumber = meetingPoint.properties.intersectNumber;
165 //TODO type ist featureCollection
166   console.log(t1);
167 
168   placeMarker(meetingPoint.geometry.coordinates, n1, t1, n2, t2,  wString, meetingsArray);
169 
170 }

317 function buttonAddMeeting(index) {
318 
319   var now = new Date();
320   let year = now.getFullYear();
321   let month = '' + (now.getMonth() + 1);
322   if(month.length < 2) { month = '0' + month; }
323   let day = '' + now.getDate();
324   if(day.length < 2) { day = '0' + day; }
325   now = day+'-'+month+'-'+year;
326 
327   var pointId = "coords";
328   pointId += index;
329    event.preventDefault();
330 
331     var newPoint = toGeoJsonPoint(meetingsArray[index].geometry.coordinates);
332 
333     var addAttr = (newPoint);
334 
335     addAttr.firstUsername = meetingsArray[index].properties.firstUsername;
336     addAttr.encounter = meetingsArray[index].properties.secondUsername;
337     addAttr.firstType = meetingsArray[index].properties.firstType;
338     addAttr.intersectNumber = meetingsArray[index].properties.intersectNumber;
339     addAttr.marker = markerText[meetingsArray[index].properties.intersectNumber];
340     addAttr.timeStamp = now;
341     var newPointString = JSON.stringify(addAttr);
342 
343     //post
344     $.ajax({
345       type: 'POST',
346       data: newPointString,
347       url: '/users/addroute',
348       contentType:"application/json"
349     }).done(function(response) {
350       console.dir(response);
351       //successful
352       if(response.error === 0) {
353         alert('Meetingpoint added: ' + response.msg);
354         readPoints();
355       }
356       else {
357         alert('Error: ' + response.msg);
358       }
359     });
