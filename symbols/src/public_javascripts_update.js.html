 11 //show routes
 12 function showRoutes() {
 13   var tableContent = "";
 14 
 15   //get JSON
 16   $.getJSON('/users/routes', function(data) {
 17     routesArray = data;
 18     console.log(data);
 19     $.each(data, function(index) {
 20       if (routesArray[index].features[0].geometry.type  == "LineString"  ) {
 21 
 22       tableContent += '<tr>';
 23       tableContent += '<td>' + index + '</td>';
 24       tableContent += '<td>' + this.name + '</td>';
 25       tableContent += '<td>' + this.routeType + '</td>';
 26       tableContent += '<td>' + this.username + '</td>';
 27       tableContent += '<td>' + this.date + '</td>';
 28       tableContent += '<td>' + this.desc + '</td>';
 29       tableContent += '<td><a href="#" class="linkeditroute" rel="' + index + '">Edit</a></td>';
 30       tableContent += '</tr>';
 31     }});
 32 
 33     //put into html
 34     $('#routeTable').html(tableContent);
 35     $('#routeTable').on('click', 'td a.linkeditroute', editRoute);
 36   });
 37 }
 38 
 39 function editRoute(event) {
 40   event.preventDefault();
 41   var id = $(this).attr('rel');
 42   var tableContent = "";
 43   tableContent += '<tr>';
 44   tableContent += '<td>' + '<input type="text" id="newName" value="' + routesArray[id].name + '"></input>' + '</td>';
 45   tableContent += '<td>' + '<input type="text" id="newType" value="' + routesArray[id].routeType + '"></input>' + '</td>';
 46   tableContent += '<td>' + '<input type="text" id="newUsername" value="' + routesArray[id].username + '"></input>' + '</td>';
 47   tableContent += '<td>' + '<input type="text" id="newDate" value="' + routesArray[id].date + '"></input>' + '</td>';
 48   tableContent += '<td>' + '<input type="text" id="newDesc" value="' + routesArray[id].desc + '"></input>' + '</td>';
 49   tableContent += '<td>' + '<td><a href="#" class="linkputroute" rel="' + routesArray[id]._id + '">Submit</a></td>';
 50   console.log(routesArray[id]._id);
 51   tableContent += '</tr>';
 52 
 53   $('#editTable').html(tableContent);
 54   $('#editTable').on('click', 'td a.linkputroute', putRoute);
