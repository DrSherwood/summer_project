  7 function loadStudy(){
  8 
  9   // jshint esversion: 6
 10   var jsonUrl = "https://www.movebank.org/movebank/service/json-auth";
 11   var study_id = document.getElementById('idInput').value; // !! add the Movebank ID for your study, available in the Study Details
 12   var individual_local_identifiers = [document.getElementById('individualInput').value]; // !! add the exact Animal IDs for the animals in the study that you want to show on the map
 13   var sensor = document.getElementById("sensorType")  .elements["sensorType"].value;
 14 // Check if a study and a animal id has been chosen
 15   if(study_id == '' && individual_local_identifiers == ''){
 16     alert("Please enter a Study and a Animal Id. For more Informations see: https://www.movebank.org/")
 17     return;
 18   }
 19   else
 20 $.getJSON(jsonUrl + "?callback=?", {
 21     study_id: study_id,
 22     individual_local_identifiers: individual_local_identifiers,
 23     //max_events_per_individual : '100',
 24     // timestamp_start: timestamp_start,
 25     // timestamp_end: timestamp_end,
 26     sensor_type: sensor
 27 // !! change if needed to specify the sensor type to display; options are gps, argos-doppler-shift, solar-geolocator, radio-transmitter, bird-ring, natural-mark
 28 }, function (data0) {
 29     data = data0;
 30     // check if the response from the sever is working according to the chosen sensor
 31     if (JSON.stringify(data0) == '{"individuals":[]}') {
 32       alert("Please try another Sensor-Type")
 33       return;
 34     }
 35     else
 36     coordinates = [];
 37     for(let i = 0; i < data0.individuals[0].locations.length; i++) {
 38       let point = [];
 39       let lat = data0.individuals[0].locations[i].location_lat;
 40       let lon = data0.individuals[0].locations[i].location_long;
 41       point.push(lon);
 42       point.push(lat);
 43       coordinates.push(point);
 44     }
 45     document.getElementById("GeoJson").value = JSON.stringify(toGeoJson(coordinates));
 46     document.getElementById("routeUsername").value = data0.individuals[0].individual_local_identifier;
 47     document.getElementById("routeDesc").value = data0.individuals[0].individual_taxon_canonical_name + ", study_id: " + data0.individuals[0].study_id;
 48     document.getElementById("routeType").value = "animal";
 49     //time
 50     var timestamp = new Date(data0.individuals[0].locations[0].timestamp);
 51     let year = timestamp.getFullYear();
 52     let month = '' + (timestamp.getMonth() + 1);
 53     if(month.length < 2) { month = '0' + month; }
 54     let day = '' + timestamp.getDate();
 55     if(day.length < 2) { day = '0' + day; }
 56     timestamp = day+'-'+month+'-'+year;
 57     document.getElementById("routeDate").value = timestamp;
 58     document.getElementById("routeName").value = data0.individuals[0].individual_taxon_canonical_name + " at " + timestamp;
 59 
 60 });
 61 
 62 }

 66 /**

 68 * @author Benjamin Rieke 408743
 69 */
 70 
 71 $(document).ready(function() {
 72   $('.btn').on('click', function() {
 73     var $this = $(this);
 74     var loadingText = 'Loading animal routes...';
 75     if ($(this).html() !== loadingText) {
 76       $this.data('original-text', $(this).html());
 77       $this.html(loadingText);
 78     }
 79     setTimeout(function() {
 80       $this.html($this.data('original-text'));
 81     }, 2000);
 82   });
 83 })

 86 /**
 87   * converts an array of coordinates into a LineString in GeoJSON format
 88   * done by using this source:
 89   * @see https://github.com/perliedman/leaflet-routing-machine/blob/344ff09c8bb94d4e42fa583286d95396d8227c65/src/L.Routing.js
 90   * @author Jonathan Bahlmann
 91   */
 92 function toGeoJson(coordinates) {
 93   return {
 94     type: 'FeatureCollection',
 95     features: [
 96       {
 97         type: 'Feature',
 98         properties: {},
 99         geometry: {
100           type: 'LineString',
101           coordinates: coordinates
102         }
103       },
104     ]
105   };
106 }

108 /**
109   * this function handles the add route button on the animal tracks webpage
110   * It is taken from the create.js function
111   * @author Jonathan Bahlmann
112   */
113 function buttonAddAnimal() {
114     event.preventDefault();
115 
116     var newRoute = document.getElementById("GeoJson").value;
117     var addAttr = JSON.parse(newRoute);
118     addAttr.name = document.getElementById("routeName").value;
119     addAttr.routeType = document.getElementById("routeType").value;
120     addAttr.username = document.getElementById("routeUsername").value;
121     addAttr.date = document.getElementById("routeDate").value;
122     addAttr.desc = document.getElementById("routeDesc").value;
123     var newRouteString = JSON.stringify(addAttr);
124 
125     //post
126     $.ajax({
127       type: 'POST',
128       data: newRouteString,
129       url: '/users/addroute',
130       contentType:"application/json"
131     }).done(function(response) {
132       console.dir(response);
133       //successful
134       if(response.error === 0) {
135         alert('route added: ' + response.msg);
136       }
137       else {
138         alert('Error: ' + response.msg);
139       }
140     });
141 }
